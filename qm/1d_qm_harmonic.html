import numpy as np
import matplotlib.pyplot as plt
from scipy.linalg import eigh_tridiagonal
In[2]:
hbar=1
m=1
L=4
omega = 1
N=1000
x=np.linspace(-L/2,L/2,N)
h=x[1]-x[0]

Harmonic Potential $$ V(x)=\frac{1}{2} m \omega^2 x^2 $$

In[3]:
V=0.5*m*omega**2*x**2
In[4]:
D2=(1/h**2)*2*np.diag(np.ones(N))+np.diag(np.ones(N-1),-1)+np.diag(np.ones(N-1),+1)

$$ -\frac{\hbar^2}{2m}\frac{d^2\psi(x)}{dx^2} + \frac{1}{2} m \omega^2 x^2 \psi(x) = E \psi(x) $$

In[5]:
H=-(hbar**2/2*m)*D2 + V
d=np.diag(H)
e=np.diag(H,-1)
In[6]:
energies,wavefuncs=eigh_tridiagonal(d,e)
In[7]:
#Normalizing wave functions
def normalize(psi,h):
    norm = np.sqrt(np.sum(np.abs(psi)**2)*h)
    return psi/norm
In[8]:
n=5#Energy State
psi=wavefuncs[:,n] #Wave function
psi_norm=normalize(psi,h) #Normalised Wave function
energy = energies[n] #nth state energy
probality_density = np.abs(psi_norm)**2
print(energy)
